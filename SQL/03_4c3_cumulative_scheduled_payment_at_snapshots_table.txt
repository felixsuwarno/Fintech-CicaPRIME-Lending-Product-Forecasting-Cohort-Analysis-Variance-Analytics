WITH lps_JOIN_installments AS
(
    SELECT
        l.loan_id,
        p.installment_no,
		l.term_months,
        l.origination_date,
        l.origination_month,
        l.snapshot_date_mob11,
        l.snapshot_date_mob12,
        p.cumulative_scheduled_payment
    FROM cica_prime."03_4c1_loan_snapshot_table" l
    JOIN cica_prime."03_4c2_payment_schedule_cumulative_table" p
        ON l.loan_id = p.loan_id
    ORDER BY l.loan_id, p.installment_no
),

lps_AGG_scheduled_payment_due_at_snapshots AS
(
    SELECT
        loan_id,
        term_months,
        origination_date,
        origination_month,
        snapshot_date_mob11,

        MAX(
            CASE
                WHEN installment_no = 11 THEN cumulative_scheduled_payment
            END
        ) AS scheduled_cumulative_pay_mob11,

        snapshot_date_mob12,		

        MAX(
            CASE
                WHEN installment_no = 12 THEN cumulative_scheduled_payment
            END
        ) AS scheduled_cumulative_pay_mob12

    FROM lps_JOIN_installments
    GROUP BY
        loan_id,
        term_months,
        origination_date,
        origination_month,
        snapshot_date_mob11,
        snapshot_date_mob12
    ORDER BY loan_id
)

SELECT *
FROM lps_AGG_scheduled_payment_due_at_snapshots
