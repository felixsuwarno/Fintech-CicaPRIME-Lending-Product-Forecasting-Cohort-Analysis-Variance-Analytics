WITH 
loans_JOIN_payment_scheduled_JOIN_cumulative_paid_snapshot AS
(
    SELECT
        l.loan_id,
        l.term_months,
        l.origination_date,
        l.origination_month,
        s.installment_no,
        s.due_date,
        s.cumulative_scheduled_payment AS scheduled_cumulative,
        l.snapshot_date_mob12,
        p.cumulative_paid_mob12
    FROM "03_4c1_loan_snapshot_table" l
    JOIN "03_4c2_contractual_payment_schedule_table" s
        ON l.loan_id = s.loan_id
    JOIN "03_4c4_cumulative_paid_at_snapshots_table" p
        ON l.loan_id = p.loan_id
    ORDER BY l.loan_id, s.installment_no
),

first_unpaid_per_loan AS
(
    SELECT
        loan_id,
        term_months,
        origination_date,
        origination_month,
        snapshot_date_mob12,

        MIN(CASE WHEN scheduled_cumulative > cumulative_paid_mob12 THEN due_date END) AS first_unpaid_due_date_mob12

    FROM loans_JOIN_payment_scheduled_JOIN_cumulative_paid_snapshot
    GROUP BY
        loan_id,
        term_months,
        origination_date,
        origination_month,
        snapshot_date_mob12
    ORDER BY loan_id
),

dpd_at_snapshots AS
(
    SELECT
        loan_id,
        term_months,
        origination_date,
        origination_month,
        snapshot_date_mob12,
        first_unpaid_due_date_mob12,

        CASE
            WHEN first_unpaid_due_date_mob12 IS NULL THEN 0
            ELSE CAST(snapshot_date_mob12 - first_unpaid_due_date_mob12 AS INTEGER)
        END AS days_past_due_mob12

    FROM first_unpaid_per_loan
    ORDER BY loan_id
)

SELECT *
FROM dpd_at_snapshots;
